Lessons learned

className statt class


render: function()
{
  return <span className="myClass">Hello World</span>
}

Doppelte Attribute


render: function()
{
  return <span className="myClass" className="myClass2">Hello World</span>
}

Keys für Arrays


render: function()
{
  var elements = [<div key="id1">Text1</div>, <div key="id2">Text2</div>];
  return <span>{elements}</span>
}

.bind


loadData: function()
{
  $.get("/jsonApi", function(data) {
    this.setState({"data": data});
  }.bind(this));
}

.bind mit Parameter


clicked: function(val)
{
  console.log(val);
},
render: function()
{
  var vals = [1, 2, 3, 4, 5];
  return vals.map(function(val) {
    return <button onClick={this.clicked.bind(this, val)}>Button {val}</button>
  }.bind(this));
}
